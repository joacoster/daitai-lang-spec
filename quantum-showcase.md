# Quantum Computing in daitai-lang v1.4

> **daitai-lang** is a formally specified language with native algebraic structures and Dirac (bra-ket) notation for quantum computing.

This example demonstrates `hilbertspace`, `observable`, and `measurement` declarations — first-class quantum primitives that compile to multiple targets.

---

## 📐 Source: `quantum-primitives.dai`

```daitai
module Quantum

import Complex, Ket, Bra, Operator from daitai.quantum

-- ─── Qubit Hilbert Space ────────────────────────────

hilbertspace Qubit over Complex:
  dimension = 2
  basis(index: Int) -> Ket:
    return Ket.standard(2, index)

-- ─── Standard Basis ─────────────────────────────────

val q0: Ket = |0⟩
val q1: Ket = |1⟩
val plus: Ket = (|0⟩ + |1⟩) |> normalize
val minus: Ket = (|0⟩ - |1⟩) |> normalize

-- ─── Observable: Spin-Z ─────────────────────────────

observable SpinZ over Qubit:
  operator() -> Operator:
    return |0⟩⟨0| - |1⟩⟨1|

  eigenvalues() -> List[Float]:
    return [1.0, -1.0]

  eigenstates() -> List[Ket]:
    return [|0⟩, |1⟩]

  expectation(state: Ket) -> Float:
    return real(⟨state|Z|state⟩)

-- ─── Projective Measurement ─────────────────────────

measurement ProjectiveZ over Qubit:
  project(state: Ket, outcome: Int) -> Ket:
    let P = outerProduct(basis(outcome), basis(outcome).dagger())
    return P.apply(state) |> normalize

  probability(state: Ket, outcome: Int) -> Float:
    return abs(⟨outcome|state⟩) |> square

  outcomes() -> List[Int]:
    return [0, 1]

-- ─── Bell State Preparation ─────────────────────────

function bellState() -> Ket:
  let initial = |0⟩ ⊗ |0⟩
  let afterH = (H ⊗ I) |> apply(initial)
  return CNOT |> apply(afterH)

-- ─── Verification ───────────────────────────────────

verify SpinZ satisfies Observable:
  hermiticity: Z.dagger() == Z
  eigendecomposition: Z.apply(|0⟩) == 1.0 * |0⟩
  completeness: sum(eigenstates() |> map(fn(e) => outerProduct(e, e.dagger()))) == I
```

## 🍎 Generated: Swift

```swift
import Accelerate
import Foundation

// MARK: - Qubit Hilbert Space

struct QubitSpace: HilbertSpace {
    typealias Scalar = Complex<Double>
    
    static let dimension: Int = 2
    
    static func basis(_ index: Int) -> Ket {
        Ket.standard(dimension: 2, index: index)
    }
}

// MARK: - Standard Basis

let q0 = Ket.basis(0)  // |0⟩
let q1 = Ket.basis(1)  // |1⟩
let plus  = (q0 + q1).normalized()   // |+⟩
let minus = (q0 - q1).normalized()   // |−⟩

// MARK: - Observable: Spin-Z

struct SpinZ: Observable {
    typealias Space = QubitSpace
    
    var operatorMatrix: Operator {
        .outerProduct(q0, q0.bra) - .outerProduct(q1, q1.bra)
    }
    
    var eigenvalues: [Double] { [1.0, -1.0] }
    
    var eigenstates: [Ket] { [q0, q1] }
    
    func expectation(_ state: Ket) -> Double {
        (state.bra * operatorMatrix * state).real
    }
}

// MARK: - Projective Measurement

struct ProjectiveZ: Measurement {
    typealias Space = QubitSpace
    typealias Outcome = Int
    
    func project(_ state: Ket, outcome: Int) -> Ket {
        let basisKet = QubitSpace.basis(outcome)
        let P = Operator.outerProduct(basisKet, basisKet.bra)
        return (P * state).normalized()
    }
    
    func probability(_ state: Ket, outcome: Int) -> Double {
        let basisKet = QubitSpace.basis(outcome)
        return (basisKet.bra * state).magnitudeSquared
    }
    
    var outcomes: [Int] { [0, 1] }
}

// MARK: - Bell State

func bellState() -> Ket {
    let initial = q0 ⊗ q0
    let afterH = (Gates.H ⊗ Operator.identity(2)).apply(initial)
    return Gates.CNOT.apply(afterH)
}

// MARK: - Verification

extension SpinZ {
    static func verify() -> [LawResult] {
        let z = SpinZ()
        return [
            LawResult("hermiticity",
                       z.operatorMatrix.dagger() == z.operatorMatrix),
            LawResult("eigendecomposition",
                       z.operatorMatrix.apply(q0) ≈ 1.0 * q0),
            LawResult("completeness",
                       z.eigenstates.map { Operator.outerProduct($0, $0.bra) }
                        .reduce(.zero, +) ≈ .identity(2)),
        ]
    }
}
```

## 🟦 Generated: TypeScript

```typescript
// Generated by daitai-tsc — TypeScript target
import { Complex, Ket, Bra, Operator, braket, outerProduct, Gates } from './quantum';

// ─── Qubit Hilbert Space ────────────────────────────

class QubitSpace {
  static readonly dimension = 2;
  
  static basis(index: number): Ket {
    return Ket.basis(2, index);
  }
}

// ─── Standard Basis ─────────────────────────────────

const q0 = Ket.basis(2, 0);  // |0⟩
const q1 = Ket.basis(2, 1);  // |1⟩
const plus  = q0.add(q1).normalize();   // |+⟩
const minus = q0.sub(q1).normalize();   // |−⟩

// ─── Observable: Spin-Z ─────────────────────────────

class SpinZ {
  get operator(): Operator {
    return outerProduct(q0, q0.dagger()).sub(
           outerProduct(q1, q1.dagger()));
  }

  get eigenvalues(): number[] { return [1.0, -1.0]; }

  get eigenstates(): Ket[] { return [q0, q1]; }

  expectation(state: Ket): number {
    return state.dagger().sandwich(this.operator, state).re;
  }
}

// ─── Projective Measurement ─────────────────────────

class ProjectiveZ {
  project(state: Ket, outcome: number): Ket {
    const basisKet = QubitSpace.basis(outcome);
    const P = outerProduct(basisKet, basisKet.dagger());
    return P.apply(state).normalize();
  }

  probability(state: Ket, outcome: number): number {
    const basisKet = QubitSpace.basis(outcome);
    return braket(basisKet.dagger(), state).absSquared();
  }

  get outcomes(): number[] { return [0, 1]; }
}

// ─── Bell State ─────────────────────────────────────

function bellState(): Ket {
  const initial = q0.tensor(q0);
  const HtensorI = Gates.H.tensor(Operator.identity(2));
  const afterH = HtensorI.apply(initial);
  return Gates.CNOT.apply(afterH);
}

// ─── Verification ───────────────────────────────────

function verifySpinZ(): { law: string; pass: boolean }[] {
  const z = new SpinZ();
  const op = z.operator;
  return [
    { law: 'hermiticity',        pass: op.isHermitian() },
    { law: 'eigendecomposition', pass: op.apply(q0).equals(q0) },
    { law: 'completeness',       pass: z.eigenstates
        .map(e => outerProduct(e, e.dagger()))
        .reduce((a, b) => a.add(b))
        .equals(Operator.identity(2), 1e-10) },
  ];
}
```

## 📦 Usage

```bash
# Parse and generate daitai output
daitai compile quantum-primitives.dai

# Generate Swift
daitai compile quantum-primitives.dai --target swift

# Generate TypeScript
daitai compile quantum-primitives.dai --target typescript

# Run algebraic verification
daitai verify quantum-primitives.dai
```
